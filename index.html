    DTT11搭配esp8266(IOT實現)

    此作品為利用DHT11感測環境的溫度與濕度，再利用esp8266來將此上傳到一個網域上，並可飲用time這個函式庫來取得目前的時間，讓使用者可以知道幾點幾分的溫度與濕度分別是多少，前提是esp8266必須先連上網。

    材料:esp8266開發版(也可用其他esp開發版，只是我認為esp8266最好用)、DHT11或DHT22、母對母杜邦線

    接線:此部分接線十分簡單，在此做簡單的敘述，先照著DHT11上的標示，將正的符號接到5V，將負的接到GND，再將中間的地方接到esp8266的4(沒有一定，只是我的習慣)，即完成

    程式碼:

    #include <ESP8266WiFi.h>
#include "DHT.h"
#include <time.h>
DHT dht(2,DHT11);
const char* ssid="your internet";
const char* password="your password";

    while(!time(nullptr)){
      Serial.print("*");
      delay(1000);
    }
    Serial.println("\nTime response.....OK");

void loop() {

    程式碼介紹:

    1.首先引入函式庫，接著變數設定。

    2.接著設定鮑率，我設115200，再來做一些腳位的初始化，設定完後會看到一個迴圈，那個迴圈是判斷eap8266是否連接到網路，如果尚未連接到網路，則會先出 ' . ' 這個符號，重複直到連接到網路當上述的動作皆完成時，就要做獲取時間的初始化，這裡我也不太清楚他是怎麼運作的所以先跳過。

    3.一切就緒後就進入loop()吧，首先為獲取時間的程式，接著為檢測是否有人進入系統所給的IP位置，如果有人進入後，我先讓程式停頓8秒，所以有一個delay(8000)，我會這樣做是因為我做過許多次實驗發現當esp8266在取得時間的過程中前幾秒為錯誤資訊，所以我讓它多跑幾秒，讓資訊變成正確的，8秒過後就為讀取溫溼度的程式，還蠻直觀的這邊比較好理解，照著函式的名稱就能理解它在做甚麼。最後要將資料傳上esp8266的IP，所以這邊會運用到一些HTML的程式，我是HTML的菜菜菜.....鳥，所以這邊的HTML程式是運用最簡單的方式撰寫，所以大致上都十分簡單，都是最簡單的標籤，所以這邊只介紹一行比較不太一樣的地方，<meta charset = \"utf-8\"></meta>，先說我會加那兩個倒斜線是因為Arduino是利用C語言的邏輯撰寫，所以我們運用雙引號要特別注意，在這邊兩個倒斜線為做跳脫字元用，而這行標籤是在為我後面的中文字做設定，加了這一行後，要再網頁裡加入中文字比較不會有亂碼出現。而我設定是30秒上船一次資料，這邊可以自己改，到這邊即大功告成了，恭喜你完成一個IOT的小實作!

    利用可變電阻控制LED燈亮度和伺服馬達

    此作品為利用可變電阻控制LED燈的亮度以及控制伺服馬達，藉由電阻大小的不同，利用類比輸入將數值輸入到Arduino nano。

    由於伺服馬達只能旋轉180度，所以需使用map將輸入值的範圍改成0到180，接著利用0～180這個數值改變伺服馬達的轉向。至於LED燈的話就不一定要更改數值輸出範圍，再利用數值的高低調整LED燈的亮度。

    材料：Arduino nano、SG-90、LED燈、杜邦線、可變電阻、麵包板

    第二題：人口遷移

    #include <iostream>
using namespace std;
int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    int r,c,k,m,max_ = -1,min_ = 2147483647;
    cin >> r >> c >> k >> m;
    int arr[r+3][c+3],temp[r+3][c+3];
    for(int i = 1;i <= r;i++){
        for(int j = 1;j <= c;j++){
            cin >> arr[i][j];
            temp[i][j] = arr[i][j];
        }
    }
    while(m > 0){
        for(int i = 1;i <= r;i++){
            for(int j = 1;j <= c;j++){
                int tmp = arr[i][j]/k;
                if(arr[i][j] == -1) continue;
                if(i != 1 && arr[i-1][j] != -1){
                    temp[i-1][j] += tmp;
                    temp[i][j] -= tmp;
                }
                if(i != r && arr[i+1][j] != -1){
                    temp[i+1][j] += tmp;
                    temp[i][j] -= tmp;
                }
                if(j != 1 && arr[i][j-1] != -1){
                    temp[i][j-1] += tmp;
                    temp[i][j] -= tmp;
                }
                if(j != c && arr[i][j+1] != -1){
                    temp[i][j+1] += tmp;
                    temp[i][j] -= tmp;
                }
            }
        }
        for(int i = 1;i <= r;i++){
            for(int j = 1;j <= c;j++){
                arr[i][j] = temp[i][j];
            }
        }
        m--;
    }
    for(int i = 1;i <= r;i++){
        for(int j = 1;j <= c;j++){
            if(arr[i][j] == -1) continue;
            if(arr[i][j] >= max_) max_ = arr[i][j];
            if(arr[i][j] <= min_) min_ = arr[i][j];
        }
    }
    cout << min_ << '\n' << max_;

    return 0;
}

        if(vec[i] < 60) fail.push_back(vec[i]);
    }
    cout << endl;
    if(pass.empty()){
        maxelement = *max_element(fail.begin(),fail.end());
        cout << maxelement << endl;
        cout << "worst case" << endl;
        return 0;
    }
    if(fail.empty()){
        minelement = *min_element(pass.begin(),pass.end());
        cout << "best case" << endl;
        cout << minelement << endl;
        return 0;
    }
    minelement = *min_element(pass.begin(),pass.end());
    maxelement = *max_element(fail.begin(),fail.end());
    if(fail.empty() == false && pass.empty() == false)
        cout << maxelement << '\n' << minelement << endl;
    return 0;
}

    #include <iostream>
#include <stack>
using namespace std;
void flip(int arr[100][100],int &r,int &c);
void change(int arr[100][100],int r,int c);//up and down
int main(){
    int r,c,m;
    while(cin >> r >> c >> m){
        int arr[100][100];
        for(int i = 0;i < r;i++){
            for(int j = 0;j < c;j++){
                cin >> arr[i][j];
            }
        }
        stack<int> s;
        for(int i = 0;i < m;i++){
            int tmp;
            cin >> tmp;
            s.push(tmp);
        }
        int l = s.size();
        for(int i = 0;i < l;i++){
            if(s.top() == 0) flip(arr,r,c);
            else change(arr,r,c);
            s.pop();
        }
        cout << r << " " << c << endl;
        for(int i = 0;i < r;i++){
            for(int j = 0;j < c;j++){
                cout << arr[i][j];
                if(j != c-1) cout << " ";
            }
            cout << endl;
        }
    }
    return 0;
}
void flip(int arr[100][100],int &r,int &c){//旋轉
    int temp[100][100];
    for(int i = 0;i < r;i++){
        for(int j = 0;j < c;j++){
            temp[c-j-1][i] = arr[i][j];
        }
    }
    for(int i = 0;i < c;i++){
        for(int j = 0;j < r;j++){
            arr[i][j] = temp[i][j];
        }
    }
    int t = c;
    c = r;
    r = t;
}
void change(int arr[100][100],int r,int c){
    int temp[100][100];
    for(int i = 0;i<r;i++){
        for(int j = 0;j < c;j++){
            temp[r-i-1][j] = arr[i][j];
        }
    }
    for(int i = 0;i < r;i++){
        for(int j = 0;j < c;j++){
            arr[i][j] = temp[i][j];
        }
    }
}

    第三題:線段覆蓋長度

    #include <iostream>
#include <algorithm>
using namespace std;
struct node{
    int a;
    int b;
};
bool cmp(node x,node y){
    if(x.a == y.a) return x.b < y.b;
    else return x.a < y.a;
}
node c[10001];
int main(){
    int n,big,small,ans = 0;
    cin >> n;
    for(int i = 0;i < n;i++)
        cin >> c[i].a >> c[i].b;
    sort(c,c+n,cmp);
    for(int i = 0;i < n;i++){
        small = c[i].a;
        big = c[i].b;
        while((i+1 < n) && c[i+1].a < big){
            if(c[i+1].b <= big){
                i++;
            }
            else{
                big = c[i+1].b;
                i++;
            }
        }
        ans = ans + big - small;
    }
    cout << ans;

    return 0;
}
